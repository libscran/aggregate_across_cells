50void compute_by_row(const tatami::Matrix<Data_, Index_>* p, const Factor_* factor, std::vector<Sum_*>& sums, std::vector<Detected_*>& detected, const Options& options) {
51 tatami::Options opt;
52 opt.sparse_ordered_index = false;
53
54 tatami::parallelize([&](size_t, Index_ s, Index_ l) {
55 auto ext = tatami::consecutive_extractor<sparse_>(p, true, s, l, opt);
56 std::vector<Sum_> tmp_sums(sums.size());
57 std::vector<Detected_> tmp_detected(detected.size());
58
59 auto NC = p->ncol();
60 std::vector<Data_> vbuffer(NC);
61 typename std::conditional<sparse_, std::vector<Index_>, Index_>::type ibuffer(NC);
62
63 for (Index_ x = s, end = s + l; x < end; ++x) {
64 auto row = [&]() {
65 if constexpr(sparse_) {
66 return ext->fetch(vbuffer.data(), ibuffer.data());
67 } else {
68 return ext->fetch(vbuffer.data());
69 }
70 }();
71
72 if (sums.size()) {
73 std::fill(tmp_sums.begin(), tmp_sums.end(), 0);
74
75 if constexpr(sparse_) {
76 for (Index_ j = 0; j < row.number; ++j) {
77 tmp_sums[factor[row.index[j]]] += row.value[j];
78 }
79 } else {
80 for (Index_ j = 0; j < NC; ++j) {
81 tmp_sums[factor[j]] += row[j];
82 }
83 }
84
85 // Computing before transferring for more cache-friendliness.
86 for (Index_ l = 0; l < tmp_sums.size(); ++l) {
87 sums[l][x] = tmp_sums[l];

89 }
90
91 if (detected.size()) {
92 std::fill(tmp_detected.begin(), tmp_detected.end(), 0);
93
94 if constexpr(sparse_) {
95 for (Index_ j = 0; j < row.number; ++j) {
96 tmp_detected[factor[row.index[j]]] += (row.value[j] > 0);
97 }
98 } else {

253Results<Sum_, Detected_> compute(const tatami::Matrix<Data_, Index_>* input, const Factor_* factor, const Options& options) {
